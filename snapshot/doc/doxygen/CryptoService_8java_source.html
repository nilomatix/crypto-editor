123  private KeyPair getKeyPair(Encryption encryption, char[] password) {
124  assert encryption.getType().isAsymmetric() : encryption.getType();
125 
126  byte[] publicKeyBytes = encryption.getPublicKey();
127  KeyPair keyPair = null;
128  if (publicKeyBytes != null) {
129  // get private key
130  PrivateKey privateKey = StoreService.getInstance().loadPrivateKey(encryption.getType().getName(), password);
131  try {
132  PublicKey publicKey = KeyFactory.getInstance(encryption.getType().getName(), "BC")
133  .generatePublic(new X509EncodedKeySpec(publicKeyBytes));
134  keyPair = new KeyPair(publicKey, privateKey);
135  } catch (Exception e) {
136  throw new CryptoEditorException("Could not recover public key from serialization", e);
137  }
138  } else {
139  // generate key pair
140  try {
141  keyPair = KeyPairGenerator.getInstance(encryption.getType().getName(), "BC").generateKeyPair();
142  // store private key in key store
143  StoreService.getInstance()
144  .storePrivateKey(encryption.getType().getName(), keyPair.getPrivate(), password, generateCertificate(keyPair));
145  // expose public key
146  encryption.setPublicKey(keyPair.getPublic().getEncoded());
147  } catch (Exception e) {
148  throw new CryptoEditorException(
149  String.format("Could not generate key-pair for algorithm %s", encryption.getType().getName()), e);
150  }
151  }
