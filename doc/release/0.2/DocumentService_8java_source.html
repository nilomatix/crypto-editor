62  private void save(Document document, File file) {
63  // map to dto
64  PersistenceDTO dto = new PersistenceDTO();
65  dto.setEncryptionType(document.getEncryption().getType());
66  dto.setEncryptionMode(document.getEncryption().getMode());
67  dto.setEncryptionPadding(document.getEncryption().getPadding());
68  // encrypt content
69  EncryptionService encryptionService = new EncryptionService();
70  encryptionService.setEncryption(document.getEncryption());
71  encryptionService.setContentInput(new ByteArrayInputStream(document.getContentBytes()));
72  encryptionService.setOnSucceeded(encryptionEvent -> {
73  // set content
74  dto.setContent(encryptionService.getValue());
75  // set hash
76  dto.setHash(cryptoService.getHash(document.getContentBytes()));
77  // set iv
78  if (document.getEncryption().getMode() != null) {
79  if (document.getEncryption().getMode().isVectorMode()) {
80  dto.setInitializationVector(document.getEncryption().getInitializationVector());
81  }
82  }
83  if (document.getEncryption().getType().isPBEType()) {
84  // set salt
85  dto.setSalt(document.getEncryption().getSalt());
86  } else if (document.getEncryption().getType().isAsymmetric()) {
87  // set public key
88  dto.setPublicKey(document.getEncryption().getPublicKey());
89  }
90  // save to file
91  SaveService saveService = new SaveService();
92  saveService.setUrl(file.getPath());
93  saveService.setPersistenceDTO(dto);
94  saveService.setOnSucceeded(saveEvent -> document.setFile(file));
95  saveService.setOnFailed(failEvent -> {
96  dialogService.showErrorDialog("Could not save file", "An error occured while saving the file");
97  throw new CryptoEditorException(saveService.getException());
98  });
99  saveService.start();
100  });
101  encryptionService.setOnFailed(failEvent -> {
102  dialogService.showErrorDialog("Could not encrypt contents", "An error occured during encryption");
103  throw new CryptoEditorException(encryptionService.getException());
104  });
105  Optional<String> optional = dialogService.showPasswordDialog();
106  if (optional.isPresent()) {
107  encryptionService.setPassword(optional.get().toCharArray());
108  encryptionService.start();
109  }
110  }

112  private Document buildDocument(PersistenceDTO dto, File file) {
113  Document document = new Document();
114  Encryption encryption = cryptoService.getEncryption(dto.getEncryptionType());
115  encryption.setMode(dto.getEncryptionMode());
116  encryption.setPadding(dto.getEncryptionPadding());
117  if (encryption.getMode() != null) {
118  if (encryption.getMode().isVectorMode()) {
119  encryption.setInitializationVector(dto.getInitializationVector());
120  }
121  }
122  if (encryption.getType().isPBEType()) {
123  encryption.setSalt(dto.getSalt());
124  } else if (encryption.getType().isAsymmetric()) {
125  encryption.setPublicKey(dto.getPublicKey());
126  }
127  document.setEncryption(encryption);
128  document.setFile(file);
129  DecryptionService decryptionService = new DecryptionService();
130  decryptionService.setCipherInput(new ByteArrayInputStream(dto.getContent()));
131  decryptionService.setEncryption(document.getEncryption());
132  decryptionService.setOnSucceeded(event -> {
133  byte[] decryptionResult = decryptionService.getValue();
134  if (!cryptoService.hashMatch(decryptionResult, dto.getHash())) {
135  dialogService.showErrorDialog("The decrypted content did not pass the hash test",
136  "The persisted hash did not match the hash value of the decrypted contents.\n" +
137  "The content might be tampered!");
138  }
139  document.setText(new String(decryptionResult));
140  });
141  decryptionService.setOnFailed(failEvent -> {
142  dialogService.showErrorDialog("Could not decrypt contents", "An error occured during decryption");
143  throw new CryptoEditorException(decryptionService.getException());
144  });
145  Optional<String> optional = dialogService.showPasswordDialog();
146  if (optional.isPresent()) {
147  decryptionService.setPassword(optional.get().toCharArray());
148  decryptionService.start();
149  }
150  return document;
151  }
